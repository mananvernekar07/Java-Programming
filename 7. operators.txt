public class Main {
    public static void main(String[] args) {

        int a = 9; //1.assignment operator
        //int b = 7;
        System.out.println(a); 

        //b*=7; //2.arithmetic operator
        int b = a% 4;//modulo operator
        System.out.println(b);

        System.out.println(3<=4); //3.comparison operator

        System.out.println(4==4 || 8<7); //4.logical operator

        System.out.println(8&10); //5.bitwise operator
    }
}
